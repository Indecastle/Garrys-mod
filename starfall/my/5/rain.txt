local EnableBetaParticleSystem = true
local BetaParticleEmitterLimit = 2

-------------------------------------------------

-- ======================== Particle Safe ========================
function RunBetaParticleSystem()
    local _emitters = {}
    local _emitterIndex = -1
    
    local particle_create_old = particle.create
    
    function initEmitters(use3D, limit)
        local max = math.min(limit - #_emitters, particle.particleEmittersLeft())
        for i=1, max do
            table.insert(_emitters, particle_create_old(owner():getPos(), false))
        end
    end
    
    function nextEmitter(origin)
        initEmitters(false, BetaParticleEmitterLimit)
        
        local length = #_emitters
        _emitterIndex = (_emitterIndex + 1) % length
        --print(length)
        local targetEmitter = _emitters[_emitterIndex+1]
        if targetEmitter then
            targetEmitter:setPos(origin)
        end
        return targetEmitter
    end
    
    particle.create = function(position, use3D)
        return nextEmitter(position)
    end
    
    local MetaParticleEmitter = getMethods("ParticleEmitter")
    MetaParticleEmitter.oldAdd = MetaParticleEmitter.add
    MetaParticleEmitter.oldDestroy = MetaParticleEmitter.destroy
    
    function MetaParticleEmitter:add(...)
        if self:getParticlesLeft() == 0 then
            --printConsole("max")
            return
        end
        return self:oldAdd(...)
    end
    
    function MetaParticleEmitter:destroy()
    end
end

if EnableBetaParticleSystem then
    RunBetaParticleSystem()
end

--if player() ~= Target then return end

local emitter = nil
local sound = nil
local m = material.load("effects/spark")


if enableSound and hasPermission("bass.loadURL", soundUrl) then
    bass.loadURL(soundUrl, "3d noblock noplay", function(snd)
        if snd then
            sound = snd
            sound:setLooping(true)
            sound:setVolume(volume)
            sound:play()
        end
    end)
end


function rain(radius, degree)
    local x =  radius * math.cos(degree)
    local y = -radius * math.sin(degree)
    local pos = Target:getPos() + Vector(x, y, startZ)
    
    if checkSky then
        local tr = trace.trace(pos, pos+Vector(0,0,1000000), nil, MASK.ALL, nil, false)
        if not tr.Hit or not tr.HitSky then
            return
        end
    end
    
    -- (Material material, Vector position, number startSize, number endSize, number startLength, number endLength, number startAlpha, number endAlpha, number dieTime)
    local part = emitter:add( m, pos, 2, 0, 20, 1, 0, 200, 1 ) -- Create a new particle at pos
    if ( part ) then
        part:setCollide(true)
        part:setBounce(0.05)
        part:setColor(Color(0,math.rand(150,200),255)*alpha)
        part:setGravity( Vector( 0, 0, -600 ) ) -- Gravity of the particle
        part:setVelocity( Vector(0,0,-50)*20 ) -- Initial velocity of the particle
    end
end

function startRain()
    emitter = particle.create(Vector(), false)
    if !isValid(emitter) then return end
    
    for i=0, forMax do
        local radius, degree = math.rand(0, maxRadius), math.rand(0, math.pi*2)
        rain(radius, degree)
    end
    --emitter:destroy()
    if emitter then emitter:destroy() end
end

timer.create("rain", 0.01, 0, function()

end)


hook.add("think", "", function()
    if enableSound and sound then
        sound:setPos(Target:getPos())
    end
end)
