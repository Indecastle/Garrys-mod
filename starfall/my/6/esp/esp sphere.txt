--@name ESP Sphere
--@author
--@client


if player() ~= owner() then return end


local IsOnlyPlayers = true 
local Radius = 10000


local hitedEnts = find.inSphere(chip():getPos(), Radius, function(ent)
    return ent ~= player()
        and (not IsOnlyPlayers or ent:getClass() == 'player')
        and (IsOnlyPlayers or ent:getClass() == 'player' or ent:getClass() == 'prop_physics')
end)

for i=1, #hitedEnts do
    if i == 199 then print("The holograms limit has been reached. (200)") break end
    ent = hitedEnts[i]
    local holo = holograms.create(chip():getPos(), Angle(), ent:getModel(), Vector(1))
    holo:setMaterial("debug/debugportals")
    holo:setColor(Color(255,255,0, 150))
    holo:setParent( ent )
    holo:addEffects( 1 )
end
