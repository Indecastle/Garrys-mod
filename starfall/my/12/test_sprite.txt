local mat0 = material.load("sprites/light_ignorez")
local mat = material.create("UnlitGeneric")
mat:setInt("$flags", 128)
mat:setTexture("$basetexture", mat0:getTexture("$basetexture"))


local HaveAlpha = true

local VertexAlpha= false
local VertexColor= false
local AlphaTest= false
local Additive= HaveAlpha
local Translucent= false

local flags = 
    (VertexColor and 16 or 0) +
    (VertexAlpha and 32 or 0) +
    (Additive and 128 or 0) +
    (AlphaTest and 256 or 0) +
    (Translucent and 2097152 or 0)
    
    
if flags ~= 0 then
    --mat:setInt("$flags", flags)    
end
--mat:recompute()


local holoSprite = hologram.create(chip():localToWorld(Vector(0,0,10)), Angle(), "models/holograms/plane.mdl")
holoSprite:setMaterial("!" .. mat:getName())
holoSprite:setSize(Vector(100, 100, 1))
holoSprite:setParent(chip())


hook.add("renderscene", "", function()
    holoSprite:setAngles((eyePos()-holoSprite:getPos()):getAngle() + Angle(90,0,0))
    
    local lightpos = holoSprite:getPos()
    local viewnormal = lightpos - eyePos()
    local distance = viewnormal:getLength()
    
    local viewdot = viewnormal:dot(chip():localToWorldAngles(Angle(90,0,0)):getForward()) / distance
    if viewdot < 0 then return end
    --printConsole(viewdot)

    local visibile = render.pixelVisible(lightpos, 16)
    local visdot = visibile * viewdot

    local color = Color(255,255,255)
    color.a = math.clamp((1000 - math.clamp(distance, 32, 800)) * visdot, 0, 100) / 100 * 255
    
    local scale = 1
    local size = math.clamp(distance * visdot * (scale or 2), 64, 512)
     
    holoSprite:setSize(Vector(size * 0.4, size * 0.4, 1))
    holoSprite:setColor(color)
end)
